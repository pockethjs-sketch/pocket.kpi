Utilities.formatDate(v, _ss().getSpreadsheetTimeZone()||'Asia/Seoul', 'yyyy-MM-dd') : v; } function _head(sh){ var lc=sh.getLastColumn(); return lc ? sh.getRange(1,1,1,lc).getValues()[0] : []; } function _rowFor(head,obj){ return head.map(function(h){ var v=obj[h]; return (v===undefined||v===null)?'':v; }); } function _read(name){ var sh=_ss().getSheetByName(name); if(!sh) return []; var rng=sh.getDataRange().getValues(); if(rng.length<2) return []; var head=rng[0], out=[]; for(var i=1;i=0){ var data=sh.getDataRange().getValues(); for(var i=1;i=1;i--){ if(String(data[i][idCol])===String(id)) sh.deleteRow(i+1); } } function _append(name,obj){ var sh=_sheet(name); var head=_head(sh); if(!head.length){ head=Object.keys(obj); sh.appendRow(head); } sh.appendRow(_rowFor(head,obj)); } function _replace(name,arr){ var sh=_sheet(name); sh.clear(); if(!arr || !arr.length) return; var head=Object.keys(arr[0]); sh.appendRow(head); var rows=arr.map(function(o){ return head.map(function(h){ var v=o[h]; return (v===undefined||v===null)?'':v; }); }); sh.getRange(2,1,rows.length,head.length).setValues(rows); } function _json(obj){ return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON); }
